  \item $10 \times v \times e - 134 = $ \dotfill
  \item $5 \times  e \times 20 = $ \dotfill
  \item $240 - 2v - 10e = $ \dotfill
  \item $e \times (2v - 206) = $ \dotfill
  \item $\dfrac{e \times v + 4}{2v + 4e} \simeq$ \dotfill
\end{itemize} \columnbreak
